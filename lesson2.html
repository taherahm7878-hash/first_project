الدرس الثاني

(2) المتغيرات

ماسوف نتعلمه في هذا الدرس
let ما هي.
const ما هي.
let و const ما الفرق بين.

let اولا : ما هي

تستخدم للأعلان عن متغيرات تتغير 

أمثلة على أشياء تتغير

عمر الأنسان.
طول الأنسان.
وزن الأنسان.
السنة الدراسية.
      وهكذا

فنلاحظ

 عمر الأنسان يتغير مع مرور الوقت وكذالك
   2023 طوله ووزنه فمثلا فلان في سنة 
   كان طوله 134سم
   ولكن في سنة 2026 اصبح طوله 144سم وبما ان المتغير (طول الأنسان , وعمره , ووزنه , وسنته الدراسية) 
 let ليس ثابتاً يتغير بمرور الوقت فنعلن عن هذا المتغير بأستخدام

أمثلة

    
 let height = 134;
 height = 144;
 

 let age = 13;
 age = 15;
 

 let weight = 25;
 weight = 24;

let لاحظ ان عند تغير قيمة المتغير نكتب اسم المتغير بدون

const ثانياً : ما هي

تستخدم للأعلان عن متغير لا يتغير

أمثلة على أشياء لا تتغير

   الرقم القومي.
   فصيلة الدم.
   عدد أيام الأسبوع.

فنلاحظ

    يوم عيد الميلاد ثابت لا يتغير وفصيلة الدم أيضاً لا تتغير
    وعدد أيام الأسبوع ثابتة لا تتغير

أمثلة

    
        const weekDays = 7;
        

        const nationalNumber = 31009022404536;

لا يمكن تغيره const لاحظ لو أعطيت للمتغير قيمة وانت أعلنت عنه بواسطة

let و const ثالثاً : الفرق بين

let عند الأعلان عن متغير يتغير نستخدم

const عند الأعلان عن متغير لا يتغير نستخدم
